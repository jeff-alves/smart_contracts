# smart_contracts

Criar ambiente:

    virtualenv venv -p python3
    source venv/bin/activate
    pip install -r requirements.txt
    python -m solc.install v0.4.21



Ativar ambiente:
	
    source venv/bin/activate
    export PATH="${PATH}:$HOME/.py-solc/solc-v0.4.21/bin"


Rodar:

    python base.py
    python base3.py
